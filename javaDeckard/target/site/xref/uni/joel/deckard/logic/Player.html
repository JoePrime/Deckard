3   import uni.joel.deckard.logic.cards.Card;
4   import java.lang.Math;
5   
6   /**
7    * Player has a name, their deck, their hand (if they are in battle), hitpoints,
8    * armor, mana and manaproduction which are all int values.
9    *
10   * @author Joel
11   */
12  public class Player {
13  
14      String name = "";
15      Deck deck;
16      Hand hand;
17      public static final int DEFAULTHITPOINTS = 50;
18      public static final int MAXHITPOINTS = 100;
19      int hitpoints = DEFAULTHITPOINTS;
20      int armor = 25;
21      int mana = 10;
22      int manaProduction = 3;
23      Player opponent;
24  
25      /**
26       * Constructor for Player. Creates a new deck for the player called "game
27       * deck" and gives the player a new hand.
28       * @param name The name that the player gets.
29       */
30      public Player(String name) {
31          this.name = name;
32          deck = new Deck("game deck");
33          hand = new Hand();
34      }
35  
36      /**
37       * Draws a new random card from the deck.
38       */
39      public void drawCard() {
40          hand.addCard(deck.newCard());
41      }
42  
43      /**
44       * Uses the specified card. Effects depend on the card.
45       *
46       * @param card The card to be used.
47       * @return True if the card was used successfully, otherwise false.
48       */
49      public boolean useCard(Card card) {
50          if (hand.amountOfCard(card) > 0) {
51              card.invoke(this);
52              hand.removeCard(card);
53              return true;
54          }
55          return false;
56      }
57  
58      /**
59       * Adds the given card to hand.
60       *
61       * @param card The card to be added to hand.
62       */
63      public void addToHand(Card card) {
64          hand.addCard(card);
65      }
66  
67      /**
68       * Adds the given card to deck.
69       *
70       * @param card The card to be added to deck.
71       */
72      public void addToDeck(Card card) {
73          deck.addCards(card, 1);
74      }
75  
76      @Override
77      public String toString() {
78          return "Player: " + name;
79      }
80  
81      public String getName() {
82          return name;
83      }
84  
85      public void setName(String name) {
86          this.name = name;
87      }
88  
89      public Deck getDeck() {
90          return deck;
91      }
92  
93      public void setDeck(Deck deck) {
94          this.deck = deck;
95      }
96  
97      public Hand getHand() {
98          return hand;
99      }
100 
101     public void setHand(Hand hand) {
102         this.hand = hand;
103     }
104 
105     public int getHitpoints() {
106         return hitpoints;
107     }
108 
109     /**
110      * Changes hitpoints to given amount. Does nothing if amount is not between
111      * 0 and Player.maxHitpoints.
112      *
113      * @param hitpoints The new value to be given to the player's hitpoints.
114      */
115     public void changeHitpointsTo(int hitpoints) {
116         if (hitpoints >= 0 && hitpoints <= MAXHITPOINTS) {
117             this.hitpoints = hitpoints;
118         }
119     }
120 
121     /**
122      * Changes hitpoints by the given amount. Amount can be positive or
123      * negative. Hitpoints will be lowered if amount is negative and vice versa.
124      * If hitpoints are tried to lower or raise too much, instead changes them
125      * to 0 or Player.maxHitpoints, respectively.
126      *
127      * @param amount The amount that the player's hitpoints should be changed
128      * by.
129      */
130     public void changeHitpointsBy(int amount) {
131         if (amount < 0) {
132             hitpoints = Math.max(0, hitpoints + amount);
133         } else {
134             hitpoints = Math.min(MAXHITPOINTS, hitpoints + amount);
135         }
136     }
137 
138     public int getArmor() {
139         return armor;
140     }
141 
142     public void setArmor(int armor) {
143         this.armor = armor;
144     }
145 
146     public int getMana() {
147         return mana;
148     }
149 
150     public void setMana(int mana) {
151         this.mana = mana;
152     }
153 
154     public int getManaProduction() {
155         return manaProduction;
156     }
157 
158     public void setManaProduction(int manaProduction) {
159         this.manaProduction = manaProduction;
160     }
161 
162     public Player getOpponent() {
163         return this.opponent;
164     }
165 
166     public void setOpponent(Player newOpponent) {
167         this.opponent = newOpponent;
168     }
169 
170 }
