1   package uni.joel.deckard.logic;

3   import java.util.Random;
4   import uni.joel.deckard.logic.cards.AttackCard;
5   import uni.joel.deckard.logic.cards.BypassAttackCard;
6   import uni.joel.deckard.logic.cards.Card;
7   import uni.joel.deckard.ui.BattleView;
8   
9   /**
10   * The main part of the game is the battle screen and actions. Those are
11   * controlled here.
12   *
13   * @author Joel
14   */
15  public class Battle {
16  
17      // add turn counter!
18      private Player p1, p2;
19      BattleView battleView;
20      int turn = 0;
21      public Player startingPlayer;
22  
23      /**
24       * Creates a new Battle Object and sets given players p1 and p2 as opponents
25       * to each other.
26       *
27       * @param p1 first player
28       * @param p2 second player
29       */
30      public Battle(Player p1, Player p2) {
31          this.p1 = p1;
32          this.p2 = p2;
33          p1.setOpponent(p2);
34          p2.setOpponent(p1);
35      }
36  
37      /**
38       * Starts the battle by creating a new BattleView runnable and running it.
39       */
40      public void StartBattle() {
41          battleView = new BattleView(this);
42          battleView.run();
43          giveStartingDeck(p1);
44          giveStartingDeck(p2);
45          giveStartingHand(p1);
46          giveStartingHand(p2);
47          battleView.updateAll();
48      }
49  
50      /**
51       * Picks a random player out of the two to start.
52       *
53       * @return Returns the starting player
54       */
55      public Player drawStartingPlayer() {
56          Random drawer = new Random();
57  //        int result = drawer.nextInt(2);       FIX THIS BACK
58          int result = 0;
59          if (result == 0) {
60              startingPlayer = p1;
61              return p1;
62          } else {
63              startingPlayer = p1;
64              return p2;
65          }
66      }
67  
68      public void giveStartingDeck(Player player) {
69          player.getDeck().addCards(new AttackCard(), 5);
70          player.getDeck().addCards(new BypassAttackCard(), 5);
71      }
72  
73      /**
74       * Gives the first five cards to the appointed player's hand at the start of
75       * the game.
76       *
77       * @param player The player the cards will be given to.
78       */
79      public void giveStartingHand(Player player) {
80          if (!player.deck.isEmpty()) {
81              for (int i = 0; i < 5; i++) {
82                  player.drawCard();
83              }
84          }
85      }
86  
87      public void endTurn() {
88          if (turn % 2 == 0) {
89              if (startingPlayer == p1) {
90                  battleView.hidePlayerCards(p1);
91              } else {
92                  battleView.hidePlayerCards(p2);
93              }
94              p1.changeManaBy(p1.getManaRecovery());
95              p2.changeManaBy(p2.getManaRecovery());
96          } else if (startingPlayer == p1) {
97              battleView.hidePlayerCards(p2);
98          } else {
99              battleView.hidePlayerCards(p1);
100         }
101         turn += 1;
102         battleView.updateAll();
103     }
104 
105     public int getTurn() {
106         return turn;
107     }
108 
109     public Player getP1() {
110         return p1;
111     }
112 
113     public Player getP2() {
114         return p2;
115     }
116 }
